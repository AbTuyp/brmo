4   package nl.b3p.brmo.bgt;
5   
6   import java.io.File;
7   import java.io.FileInputStream;
8   import java.io.FileNotFoundException;
9   import java.io.IOException;
10  import java.util.List;
11  import java.util.Properties;
12  import nl.b3p.brmo.loader.gml.BGTGMLLightLoader;
13  import org.apache.commons.logging.Log;
14  import org.apache.commons.logging.LogFactory;
15  import org.geotools.factory.GeoTools;
16  
17  public class Main {
18  
19      private static final Log LOG = LogFactory.getLog(Main.class);
20      static {
21          GeoTools.init();
22      }
23  
24      public static void main(String... args) {
25          Properties prop = new Properties();
26          try (FileInputStream fos = new FileInputStream("./bgtloader.properties")) {
27              prop.load(fos);
28              fos.close();
29          } catch (FileNotFoundException ex) {
30              LOG.fatal("Property file 'bgtloader.properties' is niet gevonden.", ex);
31              System.exit(1);
32          } catch (IOException io) {
33              LOG.fatal("Laden van de property file is mislukt.", io);
34              System.exit(1);
35          }
36  
37          LOG.warn("Let op, de commandline BGT GML Light loader maakt geen gebruik van de staging database; gebruik hiervoor de webinterface.");
38          BGTGMLLightLoader ldr = new BGTGMLLightLoader();
39          ldr.setDbConnProps(prop);
40          ldr.setScanDirectory(prop.getProperty("scandirectory"));
41  
42          List<File> zips = ldr.scanDirectory();
43          for (File zip : zips) {
44              try {
45                  int actual = ldr.processZipFile(zip);
46                  LOG.info(actual + " objecten geladen uit bestand: " + zip);
47              } catch (IOException ex) {
48                  LOG.error("Laden van bestand " + zip + " is mislukt.", ex);
49              }
50          }
51      }
52  }
