18      public static GeometryJdbcConverter getGeometryJdbcConverter(Connection conn) {
19          String databaseProductName = null;
20          try {
21              databaseProductName = conn.getMetaData().getDatabaseProductName();
22          } catch (SQLException ex) {
23              throw new UnsupportedOperationException("Cannot get database product name", ex);
24          }
25          if (databaseProductName.contains("PostgreSQL")) {
26              PostgisJdbcConverter geomToJdbc = new PostgisJdbcConverter();
27              try {
28                  // DO NOT USE conn.getSchema(). This is a JDBC 4.1 method not
29                  // supported by older PostgreSQL drivers and NOT by DBCP 1.4
30                  // used by Tomcat 7!
31                  String schema = new QueryRunner().query(conn, "select current_schema", new ScalarHandler<String>());
32                  geomToJdbc.setSchema(schema);
33              } catch (SQLException ex) {
34                  throw new UnsupportedOperationException("Cannot get/set schema: " + databaseProductName, ex);
35              }
36              return geomToJdbc;
37          } else if (databaseProductName.contains("Oracle")) {
38              try {
39                  OracleConnection oc = OracleConnectionUnwrapper.unwrap(conn);
40                  OracleJdbcConverter geomToJdbc = new OracleJdbcConverter(oc);
41                  geomToJdbc.setSchema(oc.getCurrentSchema());
42                  return geomToJdbc;
43              } catch (SQLException ex) {
44                  throw new UnsupportedOperationException("Cannot get connection: " + databaseProductName, ex);
45              }
46          } else if (databaseProductName.contains("Microsoft SQL Server")) {
47              MssqlJdbcConverter geomToJdbc = new MssqlJdbcConverter();
48              try {
49                  JtdsConnection c = MssqlConnectionUnwrapper.unwrap(conn);
50  
51                  // jTDS driver heeft geen getSchema implementatie...
52                  // ResultSetHandler<String> resultHandler = new BeanHandler<>(String.class);
53                  // String schema = new QueryRunner(geomToJdbc.isPmdKnownBroken()).query(c, "SELECT SCHEMA_NAME()", resultHandler);
54                  String schema;
55                  try (Statement s = c.createStatement(ResultSet.TYPE_SCROLL_SENSITIVE, ResultSet.CONCUR_READ_ONLY)) {
56                      // HACK we moeten een keer heen en weer anders is er geen resultaat...
57                      ResultSet rs = s.executeQuery("SELECT SCHEMA_NAME();");
58                      rs.last();
59                      int numberOfRows = rs.getRow();
60                      rs.first();
61                      schema = rs.getString(1);
62                      rs.close();
63                  }
64  
65                  if (schema != null && !schema.isEmpty()) {
66                      geomToJdbc.setSchema(schema);
67                  }
68  
69              } catch (SQLException | AbstractMethodError ex) {
70                  throw new UnsupportedOperationException("Cannot get/set schema: " + databaseProductName, ex);
71              }
72              return geomToJdbc;
73          } else {
74              throw new UnsupportedOperationException("Unknown database: " + databaseProductName);
75          }
76      }
77  }
