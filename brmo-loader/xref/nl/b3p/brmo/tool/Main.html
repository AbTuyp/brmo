129     private static void printHelp() {
130         HelpFormatter formatter = new HelpFormatter();
131         formatter.setOptionComparator(new Comparator<Option>() {
132 
133             @Override
134             public int compare(Option lhs, Option rhs) {
135                 List[] lists = new List[]{stagingDbOpts, modeOpts, outDbOpts};
136                 for (List l : lists) {
137                     int lhsIndex = l.indexOf(lhs);
138                     if (lhsIndex != -1) {
139                         return new Integer(lhsIndex).compareTo(l.indexOf(rhs));
140                     }
141                 }
142                 return lhs.getArgName().compareTo(rhs.getArgName());
143             }
144 
145         });
146         final int W = 100;
147         StringWriter sw = new StringWriter();
148         PrintWriter pw = new PrintWriter(sw);
149         formatter.printUsage(pw, 80, "java -jar brmo-loader.jar <mode> <staging-db-opties> <out-db-opties>");
150         formatter.printUsage(pw, 80, "java -cp \"brmo-loader.jar;databaseDriver.jar;lib/*\" nl.b3p.brmo.tool.Main <mode> <staging-db-opties> <out-db-opties>");
151         pw.print("\nModes:\n");
152         formatter.printOptions(pw, W, modeOptions, 2, 2);
153         pw.print("\nStaging database gegevens:\n");
154         formatter.printOptions(pw, W, stagingOptions, 2, 2);
155         pw.print("\nUitvoer database gegevens:\n");
156         formatter.printOptions(pw, W, outDbOptions, 2, 2);
157         System.out.println(sw.toString());
158     }

160     public static void main(String[] args) {
161 
162         // -stagingurl jdbc:postgresql://kx1/staging -stagingprop user=staging
163         // -stagingprop password=staging -outdburl jdbc:postgresql://kx1/rsgb
164         // -outdbprop user=rsgb -outdbprop password=rsgb -standtodb 1
165         Options options = buildOptions();
166         CommandLine cl = null;
167         try {
168             CommandLineParser parser = new PosixParser();
169 
170             cl = parser.parse(options, args);
171         } catch (ParseException e) {
172             System.out.printf("%s: %s\n\n", e.getClass().getSimpleName(), e.getMessage());
173             printHelp();
174             System.exit(1);
175         }
176 
177         int code = 0;
178         try {
179             System.out.print("Maken verbinding met Staging database... ");
180             if (cl.getOptionValue("stagingurl").contains(":oracle:")) {
181                 Class.forName("oracle.jdbc.OracleDriver");
182             }
183             
184             BasicDataSource ds = new BasicDataSource();
185             Properties stagingprops = cl.getOptionProperties("stagingprop");
186             ds.setUrl(cl.getOptionValue("stagingurl"));
187             ds.setUsername(stagingprops.getProperty("user"));
188             ds.setPassword(stagingprops.getProperty("password"));
189             ds.setConnectionProperties(cl.getOptionProperties("stagingprop").toString());
190 
191             if (cl.hasOption("list")) {
192                 code = list(ds);
193             } else if (cl.hasOption("load")) {
194                 code = load(ds, cl.getOptionValues("load"));
195             } else if (cl.hasOption("delete")) {
196                 code = delete(ds, cl.getOptionValue("delete"));
197             } else if (cl.hasOption("torsgb")) {
198                 String outDbUrl = null;
199 
200                 BasicDataSource dsOut = null;
201                 if (cl.hasOption("outdburl")) {
202                     System.out.print("Maken verbinding met RSGB database... ");
203                     outDbUrl = cl.getOptionValue("outdburl");
204                     if (outDbUrl.contains(":oracle:")) {
205                         Class.forName("oracle.jdbc.OracleDriver");
206                     }
207                     dsOut = new BasicDataSource();
208                     Properties outdbProps = cl.getOptionProperties("outdbprop");
209                     dsOut.setUrl(cl.getOptionValue("outdburl"));
210                     dsOut.setUsername(outdbProps.getProperty("user"));
211                     dsOut.setPassword(outdbProps.getProperty("password"));
212                     dsOut.setConnectionProperties(cl.getOptionProperties("outdbprop").toString());
213         
214                     System.out.println("OK");
215                 }
216 
217                 code = toRsgb(ds, dsOut);
218             }else if(cl.hasOption("getmutations")){
219                 code = getMutations(ds,cl.getOptionValues("getmutations"));
220             }
221         } catch (Exception e) {
222             e.printStackTrace(System.err);
223             code = 1;
224         }
225 
226         System.exit(code);
227     }

275     private static int getMutations(DataSource ds, String[] opts){
276         Connection con = null;
277         String dir = opts[0];
278         try {
279             con = ds.getConnection();
280 
281             Statement stmt = con.createStatement();
282             String autoProcessen = "select id FROM automatisch_proces where dtype = 'GDS2OphaalProces'";
283             ResultSet rs = stmt.executeQuery(autoProcessen);
284             while(rs.next()){
285                 Long id = rs.getLong(1);
286                 String processen = "select id,bestand_naam from laadproces where automatisch_proces = " + id;
287                 Statement laadprocesStmt = con.createStatement();
288                 ResultSet lpRs = laadprocesStmt.executeQuery(processen);
289                 while(lpRs.next()){
290                     Long lpId = lpRs.getLong(1);
291                     String bestandsNaam = lpRs.getString(2);
292                     String berichten = "select id, br_orgineel_xml, object_ref from bericht where laadprocesid = " + lpId;
293                     Statement berichtStmt = con.createStatement();
294                     ResultSet berRs = berichtStmt.executeQuery(berichten);
295                     while(berRs.next()){
296                         System.out.println("Bericht " + id + " - " + bestandsNaam);
297                         String xml = berRs.getString("br_orgineel_xml");
298                         writeXML(xml,bestandsNaam,dir);
299                     }
300                 }
301             }
302         } catch (Exception ex) {
303             System.err.println("Fout bij ophalen berichten/laadprocessen/automagische processen: " +ex.getLocalizedMessage() );
304         } finally {
305             if (con != null) {
306                 try {
307                     con.close();
308                 } catch (SQLException ex) {
309                     System.err.println("Kan verbinding niet sluiten.");
310                 }
311             }
312         }
313         return 0;
314     }

316     private static void writeXML(String xml, String filename, String directory){
317         ZipOutputStream out = null;
318         try {
319             final File f = new File(directory + "/" + filename +".zip");
320             out = new ZipOutputStream(new FileOutputStream(f));
321             ZipEntry e = new ZipEntry(filename);
322             out.putNextEntry(e);
323             byte[] data = xml.getBytes();
324             out.write(data, 0, data.length);
325             out.closeEntry();
326             out.close();
327         } catch (FileNotFoundException ex) {
328             System.err.println("File not found" + ex.getLocalizedMessage());
329         } catch (IOException ex) {
330             System.err.println("IOException " + ex.getLocalizedMessage());
331         } finally {
332             try {
333                 out.close();
334             } catch (IOException ex) {
335                 Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
336             }
337         }
338 
339     }
