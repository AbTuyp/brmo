32      public SynchroonAntwoordBericht beantwoordSynchroneVraag(VraagBericht vraag) throws StUFFout {
33          try {
34              LOG.debug("Er is antwoord ontvangen van soort: " + vraag.getStuurgegevens().getBerichtsoort());
35              SynchroonAntwoordBericht antw = new SynchroonAntwoordBericht();
36              antw.setStuurgegevens(vraag.getStuurgegevens());
37              return antw;
38          } catch (Exception e) {
39              FoutBericht fout = StUFbg204Util.maakFout();
40              throw new StUFFout("Not implemented yet.", fout, e);
41          }
42      }
