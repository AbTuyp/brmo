4   package nl.b3p.brmo.test.util.database.dbunit;
5   
6   import java.sql.SQLException;
7   import org.dbunit.DatabaseUnitException;
8   import org.dbunit.database.IDatabaseConnection;
9   import org.dbunit.dataset.DefaultDataSet;
10  import org.dbunit.dataset.DefaultTable;
11  import org.dbunit.operation.DatabaseOperation;
12  
13  /**
14   * DBunit utility om databases leeg te maken.
15   *
16   * @author mprins
17   */
18  public final class CleanUtil {
19  
20      /**
21       * leegt de BRK tabellen in het RSGB schema. kan worden gebruikt in een
22       * {@code @After} van een test case.
23       *
24       * @param rsgb database welke geleegd moet worden
25       * @throws org.dbunit.DatabaseUnitException
26       * @throws java.sql.SQLException als er iets misgaat in de database
27       */
28      public static void cleanRSGB(final IDatabaseConnection rsgb) throws DatabaseUnitException, SQLException {
29          /* cleanup rsgb, doet:
30                  DELETE FROM brondocument;
31                  DELETE FROM herkomst_metadata;
32                  DELETE FROM zak_recht;
33                  DELETE FROM ingeschr_niet_nat_prs;
34                  ...
35                  DELETE FROM kad_perceel_archief;
36                  DELETE FROM kad_perceel;
37                  DELETE FROM kad_onrrnd_zk_archief;
38                  DELETE FROM kad_onrrnd_zk;
39           dus omgekeerde volgorde tov. onderstaande array
40           */
41          DatabaseOperation.DELETE_ALL.execute(rsgb, new DefaultDataSet(new DefaultTable[]{
42              new DefaultTable("kad_onrrnd_zk"),
43              new DefaultTable("kad_onrrnd_zk_his_rel"),
44              new DefaultTable("kad_onrrnd_zk_aantek"),
45              new DefaultTable("kad_onrrnd_zk_archief"),
46              new DefaultTable("kad_perceel"),
47              new DefaultTable("kad_perceel_archief"),
48              new DefaultTable("subject"),
49              new DefaultTable("prs"),
50              new DefaultTable("nat_prs"),
51              new DefaultTable("ingeschr_nat_prs"),
52              new DefaultTable("niet_ingezetene"),
53              new DefaultTable("ander_nat_prs"),
54              new DefaultTable("niet_nat_prs"),
55              new DefaultTable("ingeschr_niet_nat_prs"),
56              new DefaultTable("app_re"),
57              new DefaultTable("app_re_archief"),
58              new DefaultTable("zak_recht"),
59              new DefaultTable("zak_recht_aantek"),
60              new DefaultTable("herkomst_metadata"),
61              new DefaultTable("brondocument")}
62          ));
63      }
64  
65      /**
66       * leegt de bericht, laadproces en job tabellen in het statging schema. kan
67       * worden gebruikt in een {@code @After} van een test case.
68       *
69       * @param staging database welke geleegd moet worden
70       * @throws org.dbunit.DatabaseUnitException
71       * @throws java.sql.SQLException als er iets misgaat in de database
72       */
73      public static void cleanSTAGING(final IDatabaseConnection staging) throws DatabaseUnitException, SQLException {
74          DatabaseOperation.DELETE_ALL.execute(staging, new DefaultDataSet(new DefaultTable[]{
75              new DefaultTable("laadproces"),
76              new DefaultTable("bericht"),
77              new DefaultTable("job")}));
78      }
79  
80      /**
81       * private by design.
82       */
83      private CleanUtil() {
84      }
85  }
