2   package nl.b3p.brmo.loader.jdbc;
3   
4   import com.vividsolutions.jts.io.ParseException;
5   import java.sql.SQLException;
6   import org.postgis.PGgeometry;
7   
8   /**
9    *
10   * @author Matthijs Laan
11   */
12  public class PostgisJdbcConverter extends GeometryJdbcConverter {
13  
14      private String schema = null;
15  
16      @Override
17      public boolean isDuplicateKeyViolationMessage(String message) {
18          return message!=null && message.startsWith("ERROR: duplicate key value violates unique constraint");
19      }
20  
21      @Override
22      public String createPSGeometryPlaceholder() throws SQLException {
23          //return "ST_GeomFromText(?, 28992)";
24          return "?";
25      }
26     
27      @Override
28      public Object convertToNativeGeometryObject(String param) throws SQLException, ParseException {
29          //return param;
30          if (param == null || param.trim().length() == 0) {
31              return null;
32          }
33          return new PGgeometry("SRID=28992;" + param);
34      }
35  
36      @Override
37      public String getSchema() {
38          return schema;
39      }
40  
41      public void setSchema(String schema) {
42          this.schema = schema;
43      }
44  
45      @Override
46      public String getGeomTypeName() {
47          return "geometry";
48      }
49      
50      @Override
51      public String buildPaginationSql(String sql, int offset, int limit) {
52          StringBuilder builder = new StringBuilder(sql);
53          builder.append(" LIMIT ");
54          builder.append(limit);
55          builder.append(" OFFSET ");
56          builder.append(offset);
57          return builder.toString();
58      }
59  
60      @Override
61      public boolean useSavepoints() {
62          return true;
63      }
64  
65       @Override
66      public boolean isPmdKnownBroken() {
67          return false;
68      }
69      @Override
70      public String getGeotoolsDBTypeName() {
71          return "postgis";
72      }
73  }
